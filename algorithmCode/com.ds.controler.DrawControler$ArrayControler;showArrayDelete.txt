		Scanner scan = new Scanner(System.in);
		int n = scan.nextInt();
		int[] array = new int[n+1];
		for(int i=0; i<n; ++i)
			array[i] = scan.nextInt();
		//»ñÈ¡É¾³ýÎ»ÖÃ
		int p = scan.nextInt();

		for(int i=0; i<n-1; ++i){
			if(i!=0)
				System.out.print(" ");
			System.out.print(array[i]);
		}
	}
}
