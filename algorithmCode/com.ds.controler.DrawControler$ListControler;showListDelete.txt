class List{
	public int x;
	public List next;
	
	public List(int x){
		this.x = x;
		next = null;
	}
}

public class Main {
	public static void main(String[] args){
		Scanner scan = new Scanner(System.in);
		int n = scan.nextInt();
		List head = new List(0), pre = head;
		for(int i=0; i<n; ++i){
			List cur = new List(scan.nextInt());
			pre.next = cur;
			pre = cur;
		}
		//»ñÈ¡É¾³ýÎ»ÖÃ
		int p = scan.nextInt();
		
		int pp = 0;
		List cur = head.next;
		pre = head;
		while(pp < p) {
			pre = cur;
			cur = cur.next;
			++pp;
		}
		
		pre.next = cur.next;
		
		for(pre = head.next; pre != null; pre = pre.next) {
			if(pre != head.next)
				System.out.print(" ");
			System.out.print(pre.x);
		}
	}
}
